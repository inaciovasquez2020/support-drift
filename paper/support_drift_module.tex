\section{Support Drift Obstruction (Vasquez, 2026)}
\begin{lemma}[Support Drift Obstruction]
For the finitary local permutation group $\Gamma_{0}(R)$, Uniform Local Fragmentation fails.
Consequently $\operatorname{cw}(\Gamma_{0}(R))=\infty$ and the Non-Abelian Subadditivity inequality
cannot hold with any uniform constant.
\end{lemma}
